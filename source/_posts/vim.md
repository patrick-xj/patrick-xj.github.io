```bash 一般模式
# 光标跳转
	h				# 左
	l				# 右
	j				# 下
	k				# 上
# 单词间跳转
	w				# 下一个单词词首
	e				# 当前单词或下一个单词词尾
	b				# 返回到当前单词或下一个单词词首
# 句子间跳转
	)				# 下一句
	(				# 前一句
# 段落间跳转
	}				# 下一段
	{				# 前一段
# 行间跳转
	gg				# 跳转至开头;等同于1G
	G				# shift+G,跳转至行尾
	H				# 光标移动到当前页第一行的第一个字符
	M				# 光标移动到当前页的中央的第一个字符
	L				# 光标移动到当前页的最后行第一个字符
# 行内跳转
 	^				# 所在行的行首;第一个非空白字符
 	$				# 所在行的绝对行尾
 	0				# 所在行的绝对行首
# 快捷键使用
	u        			# 撤销
	Ctrl + r 			# 反撤销
	2+<space>			# 光标向后移动2个字符
# 搜索
    /word 			 	# 光标所在字符向下查找word字符
    ？word 				# 光标所在字符向上查找word字符
        n 			 	# 重复前一个查找动作
        N 			 	# 与n相反进行查找
# 新增
	o				# 向下新增一行
	O				# 向上新增一行
	x				# 向后删除一个字符
	X				# 向前删除一个字符
# 删除
	x  				# 删除单个字符
	2x 				# 删除光标所在向3个字符
	dw 				# 删除当前光标所在单词并跳至下一个单词词首
	dd 				# 删除光标所在行
	5dd				# 删除光标所在行并向下5行
	d$ 				# 删除光标所在字符到行结尾
	d0 				# 删除光标所在字符到绝对行首
	d^ 				# 删除光标所在字符到行首首字符处
# 复制
	yy 				# 复制光标所在行
	5yy				# 复制光标所在行向下的5行
	 . 				# 重复前一个动作
	p  				# 将复制的数据粘贴到光标所在的下一行
	P  				# 将复制的数据粘贴到光标所在的上一行
	r  				# 只会取代光标所在的单个字符
	R  				# 一直取代光标所在的字符,知道结尾
# 区块选项
	v        			# 字符选择;配合左右键使用
	V        			# 选择行;配合上下键使用
	Ctrl + v 			# 选择区块;配合上下左右键使用
```

```bash 末行模式
# 末行模式下
	:w                        # 保存
	:q                        # 退出
	:!                        # 强制
	:x                        # 保存退出
	:qall                     # 退出所有
	:1,7d                     # 删除1到7行
	%d                        # 删除所有行
	:set all                  # 显示目前所有环节参数指定值
	:set nu                   # 显示行号
	:set nonu                 # 取消行号显示
	:set backup               # 自动备份文件,源文件另存为file~
	:syntax on                # 显示颜色
	:syntax off               # 取消颜色显示
	:set paste                # 不会自动缩进;纯粘贴模式
	:set nopaste              # 自动缩进,取消纯粘贴模式
# 替换、插入
	:5,$s/test/1234/g         # 从第5行到最后一行将test替换为1234
	:5,$s/test/1234/gc        # 每一个替换字符确认是否替换
	:%s/|//g                  # 删除所有|字符
	:%s/\s/|/g                # 所有空格转换为|
	:%s/^/|/g                 # 每行行首字符前添加|
	:%s/$/|/g                 # 每行行尾字符后添加|
```

```bash 插入模式
# 进入配置文件
vim [options] file_name
    +5                            # 表示直接打开文件后到第5行
vim [options] file1 file2 ...     # 一次打开多个文件
    :next                         # 跳转到下一个文件
    :first                        # 回到第一个文件
    :last                         # 调到最后一个
    :prev                         # 返回到前一个
vim [options] file1 file2 ...   
    -o                            # 一次将多个文件打开横向显示
    -O                            # 一次性将多个文件竖立显示
    # Control + w 松开 然后按上下左右跳转至下一界面
```
